// ==================== AI FEEDBACK BADGE ====================
const AIFeedbackBadge = ({ feedback, compact = false }) => {
  if (!feedback) return null;

  const getScoreColor = (score) => {
    if (score >= 0.85) return 'bg-green-100 text-green-800 border-green-300';
    if (score >= 0.7) return 'bg-yellow-100 text-yellow-800 border-yellow-300';
    return 'bg-red-100 text-red-800 border-red-300';
  };

  if (compact) {
    return (
      <div className="inline-flex items-center gap-1 px-2 py-1 text-xs rounded border bg-purple-50 text-purple-700 border-purple-200">
        <Brain className="w-3 h-3" />
        <span>AI: {(feedback.score * 100).toFixed(0)}%</span>
      </div>
    );
  }

  return (
    <div className={`p-4 rounded-lg border ${getScoreColor(feedback.score)}`}>
      <div className="flex items-start gap-3">
        <Brain className="w-5 h-5 flex-shrink-0 mt-0.5" />
        <div className="flex-1">
          <div className="flex items-center gap-2 mb-2">
            <span className="font-semibold">AI –û—Ü–µ–Ω–∫–∞:</span>
            <span className="text-2xl font-bold">{(feedback.score * 100).toFixed(0)}%</span>
          </div>
          
          <p className="text-sm mb-3">{feedback.feedback}</p>
          
          {feedback.matched_keywords && feedback.matched_keywords.length > 0 && (
            <div className="mb-2">
              <p className="text-xs font-medium mb-1">–ù–∞–π–¥–µ–Ω–Ω—ã–µ –∫–ª—é—á–µ–≤—ã–µ —Å–ª–æ–≤–∞:</p>
              <div className="flex flex-wrap gap-1">
                {feedback.matched_keywords.map((kw, idx) => (
                  <span key={idx} className="px-2 py-0.5 text-xs bg-white rounded-full">
                    {kw}
                  </span>
                ))}
              </div>
            </div>
          )}
          
          {feedback.suggestions && feedback.suggestions.length > 0 && (
            <div>
              <p className="text-xs font-medium mb-1">–†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–∏:</p>
              <ul className="text-xs space-y-1">
                {feedback.suggestions.map((suggestion, idx) => (
                  <li key={idx}>‚Ä¢ {suggestion}</li>
                ))}
              </ul>
            </div>
          )}
          
          <div className="mt-2 text-xs opacity-70">
            –ú–µ—Ç–æ–¥: {feedback.method === 'ai' ? 'AI –∞–Ω–∞–ª–∏–∑' : '–ö–ª—é—á–µ–≤—ã–µ —Å–ª–æ–≤–∞'}
          </div>
        </div>
      </div>
    </div>
  );
};

// ==================== REGRADE BUTTON ====================
const RegradeButton = ({ answerId, onSuccess }) => {
  const [loading, setLoading] = useState(false);
  const [success, setSuccess] = useState(false);

  const handleRegrade = async () => {
    setLoading(true);
    
    try {
      const response = await fetch(`/api/ai-grading/answers/${answerId}/regrade`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${localStorage.getItem('token')}`
        },
        body: JSON.stringify({ use_stricter_threshold: false })
      });
      
      const result = await response.json();
      
      setSuccess(true);
      setTimeout(() => setSuccess(false), 3000);
      
      if (onSuccess) onSuccess(result);
      
    } catch (error) {
      console.error('Regrade failed:', error);
      alert('–ù–µ —É–¥–∞–ª–æ—Å—å –ø–µ—Ä–µ–æ—Ü–µ–Ω–∏—Ç—å –æ—Ç–≤–µ—Ç');
    } finally {
      setLoading(false);
    }
  };

  return (
    <button
      onClick={handleRegrade}
      disabled={loading}
      className="flex items-center gap-2 px-3 py-2 text-sm bg-purple-600 text-white rounded-lg hover:bg-purple-700 disabled:opacity-50 transition"
    >
      {loading ? (
        <>
          <Loader2 className="w-4 h-4 animate-spin" />
          –ü–µ—Ä–µ–æ—Ü–µ–Ω–∫–∞...
        </>
      ) : success ? (
        <>
          <CheckCircle className="w-4 h-4" />
          –ì–æ—Ç–æ–≤–æ!
        </>
      ) : (
        <>
          <RefreshCw className="w-4 h-4" />
          –ü–µ—Ä–µ–æ—Ü–µ–Ω–∏—Ç—å —Å AI
        </>
      )}
    </button>
  );
};

// ==================== PLAGIARISM CHECK ====================
const PlagiarismCheck = ({ sessionId }) => {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState(null);

  const checkPlagiarism = async () => {
    setLoading(true);
    
    try {
      const response = await fetch(
        `/api/ai-grading/sessions/${sessionId}/check-plagiarism?threshold=0.8`,
        {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${localStorage.getItem('token')}`
          }
        }
      );
      
      const data = await response.json();
      setResult(data);
      
    } catch (error) {
      console.error('Plagiarism check failed:', error);

          <h3 className="font-semibold">–ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø–ª–∞–≥–∏–∞—Ç</h3>

          onClick={checkPlagiarism}
          disabled={loading}

        >
          {loading ? (
            <>
              <Loader2 className="w-4 h-4 animate-spin" />
