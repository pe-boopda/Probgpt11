// ==================== STATS CARDS ====================
const StatsCard = ({ icon: Icon, label, value, change, color = "blue" }) => {
  const colors = {
    blue: "bg-blue-100 text-blue-600",
    green: "bg-green-100 text-green-600",
    red: "bg-red-100 text-red-600",
    yellow: "bg-yellow-100 text-yellow-600"
  };

  return (
    <div className="bg-white p-6 rounded-lg shadow">
      <div className="flex items-center justify-between mb-4">
        <div className={`p-3 rounded-lg ${colors[color]}`}>
          <Icon className="w-6 h-6" />
        </div>
        {change && (
          <div className={`flex items-center gap-1 text-sm ${
            change > 0 ? 'text-green-600' : 'text-red-600'
          }`}>
            <TrendingUp className={`w-4 h-4 ${change < 0 ? 'rotate-180' : ''}`} />
            <span>{Math.abs(change)}%</span>
          </div>
        )}
      </div>
      <div className="text-2xl font-bold text-gray-900">{value}</div>
      <div className="text-sm text-gray-600">{label}</div>
    </div>
  );
};

// ==================== QUESTION ANALYTICS ====================
const QuestionAnalyticsCard = ({ question, expanded, onToggle }) => {
  const getDifficultyColor = (difficulty) => {
    switch (difficulty) {
      case 'easy': return 'bg-green-100 text-green-800';
      case 'medium': return 'bg-yellow-100 text-yellow-800';
      case 'hard': return 'bg-red-100 text-red-800';
      default: return 'bg-gray-100 text-gray-800';
    }
  };

  return (
    <div className="bg-white rounded-lg shadow border-l-4 border-blue-500">
      {/* Header */}
      <div 
        className="p-4 cursor-pointer hover:bg-gray-50"
        onClick={onToggle}
      >
        <div className="flex items-start justify-between">
          <div className="flex-1">
            <div className="flex items-center gap-3 mb-2">
              <span className={`px-3 py-1 rounded-full text-xs font-medium ${getDifficultyColor(question.difficulty)}`}>
                {question.difficulty === 'easy' ? 'Лёгкий' : 
                 question.difficulty === 'medium' ? 'Средний' : 'Сложный'}
              </span>
              <span className="text-sm text-gray-600">
                Вопрос #{question.id} • {question.points} баллов
              </span>
            </div>
            <p className="font-medium text-gray-900 mb-2">{question.text}</p>
            
            <div className="flex items-center gap-6 text-sm">
              <div className="flex items-center gap-2">
                <CheckCircle className="w-4 h-4 text-green-600" />
                <span className="text-gray-700">
                  Правильно: <strong>{question.correct_count}</strong>
                </span>
              </div>
              <div className="flex items-center gap-2">
                <XCircle className="w-4 h-4 text-red-600" />
                <span className="text-gray-700">
                  Неправильно: <strong>{question.incorrect_count}</strong>
                </span>
              </div>
              <div className="flex items-center gap-2">
                <BarChart3 className="w-4 h-4 text-blue-600" />
                <span className="text-gray-700">
                  Процент: <strong>{question.percentage.toFixed(1)}%</strong>
                </span>
              </div>
            </div>
          </div>
          
          <div className="ml-4">
            {expanded ? 
              <ChevronDown className="w-5 h-5 text-gray-400" /> : 
              <ChevronRight className="w-5 h-5 text-gray-400" />
            }
          </div>
        </div>

        {/* Progress Bar */}
        <div className="mt-4 h-2 bg-gray-200 rounded-full overflow-hidden">
          <div 
            className={`h-full transition-all ${
              question.percentage >= 80 ? 'bg-green-500' :
              question.percentage >= 50 ? 'bg-yellow-500' : 'bg-red-500'
            }`}
            style={{ width: `${question.percentage}%` }}
          />
        </div>
      </div>

      {/* Expanded Details */}
      {expanded && (
        <div className="px-4 pb-4 border-t">
          <div className="pt-4">
            <h4 className="font-semibold text-gray-900 mb-3">
              Частые ошибки:
            </h4>
            
            {question.common_mistakes.length > 0 ? (
              <div className="space-y-3">
                {question.common_mistakes.map((mistake, idx) => (
                  <div key={idx} className="p-3 bg-red-50 border border-red-200 rounded-lg">
                    <div className="flex items-start gap-3">
                      <AlertTriangle className="w-5 h-5 text-red-600 flex-shrink-0 mt-0.5" />
                      <div className="flex-1">
                        <div className="flex items-center justify-between mb-1">
                          <span className="font-medium text-red-900">
                            Ответ: "{mistake.answer}"
                          </span>
                          <span className="text-sm text-red-700">
                            {mistake.count} студентов
                          </span>
                        </div>
                        <p className="text-sm text-red-800">
                          Причина: {mistake.reason}
                        </p>
                      </div>
                    </div>
                  </div>
                ))}
              </div>
            ) : (
              <p className="text-sm text-gray-600">
                Все студенты ответили правильно!
              </p>
            )}

            {question.percentage < 70 && (
              <div className="mt-4 p-3 bg-yellow-50 border border-yellow-200 rounded-lg">
                <div className="flex items-start gap-2">
                  <Brain className="w-5 h-5 text-yellow-600 flex-shrink-0 mt-0.5" />
                  <div>
                    <p className="font-semibold text-yellow-900 mb-1">
                      Рекомендация:
                    </p>
                    <p className="text-sm text-yellow-800">
                      Этот вопрос вызвал затруднения у многих студентов. 
                      Рекомендуется пересмотреть формулировку или провести 
                      дополнительное объяснение этой темы на следующем занятии.
                    </p>
                  </div>
                </div>
              </div>
            )}
          </div>
        </div>
      )}
    </div>
  );
};

// ==================== MAIN DASHBOARD ====================
const TeacherAnalyticsDashboard = () => {
  const [data] = useState(mockTestData);
  const [expandedQuestions, setExpandedQuestions] = useState(new Set());
  const [filter, setFilter] = useState('all'); // all, passed, failed

  const toggleQuestion = (questionId) => {
    const newExpanded = new Set(expandedQuestions);
    if (newExpanded.has(questionId)) {
      newExpanded.delete(questionId);
    } else {
      newExpanded.add(questionId);
    }
    setExpandedQuestions(newExpanded);
  };

  // Calculate stats
  const totalStudents = data.sessions.length;
  const passedStudents = data.sessions.filter(s => s.passed).length;
  const failedStudents = totalStudents - passedStudents;
  const avgScore = data.sessions.reduce((sum, s) => sum + s.percentage, 0) / totalStudents;
  const suspiciousCount = data.sessions.filter(s => s.suspicious_activity > 0).length;

  // Filter sessions
  const filteredSessions = data.sessions.filter(s => {
    if (filter === 'passed') return s.passed;
    if (filter === 'failed') return !s.passed;
    return true;
  });

  return (
    <div className="min-h-screen bg-gray-50 p-8">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="mb-8">
          <div className="flex items-center justify-between mb-4">
            <div>
              <h1 className="text-3xl font-bold text-gray-900 mb-2">
                Аналитика теста
              </h1>
              <p className="text-gray-600">{data.test.title}</p>
            </div>
            
            <button className="flex items-center gap-2 px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700">
              <Download className="w-4 h-4" />
              Экспорт отчёта
            </button>
          </div>
        </div>

        {/* Stats Overview */}
        <div className="grid grid-cols-1 md:grid-cols-4 gap-6 mb-8">
          <StatsCard
            icon={Users}
            label="Всего студентов"
            value={totalStudents}
            color="blue"
          />
          <StatsCard
            icon={CheckCircle}
            label="Сдали тест"
            value={passedStudents}
            change={12}
            color="green"
          />
          <StatsCard
            icon={BarChart3}
            label="Средний балл"
            value={`${avgScore.toFixed(1)}%`}
            change={5}
            color="blue"
          />
          <StatsCard
            icon={AlertTriangle}
            label="Подозрительная активность"
            value={suspiciousCount}
            color="yellow"
          />
        </div>

        {/* Question Analytics */}
        <div className="mb-8">
          <div className="flex items-center justify-between mb-4">
            <h2 className="text-2xl font-bold text-gray-900">
              Анализ вопросов
            </h2>
            <button
              onClick={() => {
                if (expandedQuestions.size === data.questionAnalytics.length) {
                  setExpandedQuestions(new Set());
                } else {
                  setExpandedQuestions(new Set(data.questionAnalytics.map(q => q.id)));
                }
              }}
              className="text-sm text-blue-600 hover:text-blue-700 font-medium"
            >
              {expandedQuestions.size === data.questionAnalytics.length ? 
                'Свернуть все' : 'Развернуть все'}
            </button>
          </div>

          <div className="space-y-4">
            {data.questionAnalytics.map((question) => (
              <QuestionAnalyticsCard
                key={question.id}
                question={question}
                expanded={expandedQuestions.has(question.id)}
                onToggle={() => toggleQuestion(question.id)}
              />
            ))}
          </div>
        </div>

        {/* Student Performance */}
        <div className="mb-8">
          <h2 className="text-2xl font-bold text-gray-900 mb-4">
            Матрица ответов
          </h2>
          <StudentPerformanceTable data={data.studentPerformance} />
        </div>

        {/* Insights */}

          <h2 className="text-xl font-bold text-gray-900 mb-4">
            Аналитические выводы
          </h2>
          
          <div className="space-y-4">
            <div className="p-4 bg-green-50 border border-green-200 rounded-lg">
              <div className="flex items-start gap-3">
                <CheckCircle className="w-5 h-5 text-green-600 flex-shrink-0 mt-0.5" />
                <div>
                  <p className="font-semibold text-green-900 mb-1">
                    Хорошие результаты
                  </p>
                  <p className="text-sm text-green-800">
                    Вопрос #1 показал отличные результаты - 93.3% студентов ответили правильно.
                    Эта тема хорошо усвоена.
                  </p>
                </div>
              </div>
            </div>

            <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
              <div className="flex items-start gap-3">

};
