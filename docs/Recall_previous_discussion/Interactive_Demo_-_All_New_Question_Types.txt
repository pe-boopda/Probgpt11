// ==================== MATCHING QUESTION ====================
const MatchingQuestionDemo = () => {
  const [matches, setMatches] = useState({});
  const [selectedLeft, setSelectedLeft] = useState(null);

  const leftItems = [
    { id: 'left_1', text: '–§—Ä–∞–Ω—Ü–∏—è' },
    { id: 'left_2', text: '–ì–µ—Ä–º–∞–Ω–∏—è' },
    { id: 'left_3', text: '–ò—Ç–∞–ª–∏—è' },
    { id: 'left_4', text: '–ò—Å–ø–∞–Ω–∏—è' },
  ];

  const rightItems = [
    { id: 'right_1', text: '–ü–∞—Ä–∏–∂' },
    { id: 'right_2', text: '–ë–µ—Ä–ª–∏–Ω' },
    { id: 'right_3', text: '–†–∏–º' },
    { id: 'right_4', text: '–ú–∞–¥—Ä–∏–¥' },
  ];

  const correctMatches = {
    'left_1': 'right_1',
    'left_2': 'right_2',
    'left_3': 'right_3',
    'left_4': 'right_4',
  };

  const handleLeftClick = (leftId) => {
    setSelectedLeft(selectedLeft === leftId ? null : leftId);
  };

  const handleRightClick = (rightId) => {
    if (!selectedLeft) return;

    const newMatches = { ...matches };
    Object.keys(newMatches).forEach((key) => {
      if (newMatches[key] === rightId) delete newMatches[key];
    });
    newMatches[selectedLeft] = rightId;
    
    setMatches(newMatches);
    setSelectedLeft(null);
  };

  const handleRemoveMatch = (leftId) => {
    const newMatches = { ...matches };
    delete newMatches[leftId];
    setMatches(newMatches);
  };

  const checkAnswers = () => {
    let correct = 0;
    Object.keys(correctMatches).forEach((left) => {
      if (matches[left] === correctMatches[left]) correct++;
    });
    return { correct, total: Object.keys(correctMatches).length };
  };

  const getRightItemById = (id) => rightItems.find((item) => item.id === id);
  const isRightUsed = (id) => Object.values(matches).includes(id);

  const result = checkAnswers();

  return (
    <div className="space-y-4">
      <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
        <p className="text-sm text-blue-900">
          üí° –ö–ª–∏–∫–Ω–∏—Ç–µ –Ω–∞ —ç–ª–µ–º–µ–Ω—Ç —Å–ª–µ–≤–∞, –∑–∞—Ç–µ–º –Ω–∞ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π —ç–ª–µ–º–µ–Ω—Ç —Å–ø—Ä–∞–≤–∞
        </p>
      </div>

      <div className="grid grid-cols-2 gap-6">
        {/* Left Column */}
        <div className="space-y-3">
          <h3 className="text-sm font-medium text-gray-700">–°—Ç—Ä–∞–Ω—ã:</h3>
          {leftItems.map((item) => {
            const isSelected = selectedLeft === item.id;
            const hasMatch = matches[item.id];
            const matchedRight = hasMatch ? getRightItemById(matches[item.id]) : null;

            return (
              <div key={item.id}>
                <button
                  onClick={() => handleLeftClick(item.id)}
                  className={`w-full p-4 rounded-lg border-2 text-left transition ${
                    isSelected
                      ? 'border-blue-500 bg-blue-50 shadow-md'
                      : hasMatch
                      ? 'border-green-300 bg-green-50'
                      : 'border-gray-300 bg-white hover:border-gray-400'
                  }`}
                >
                  <div className="flex items-center justify-between">
                    <span className="font-medium">{item.text}</span>
                    {hasMatch && (
                      <button
                        onClick={(e) => {
                          e.stopPropagation();
                          handleRemoveMatch(item.id);
                        }}
                        className="text-red-600 hover:text-red-700"
                      >
                        ‚úï
                      </button>
                    )}
                  </div>
                </button>
                
                {matchedRight && (
                  <div className="mt-2 flex items-center gap-2 text-sm text-green-700">
                    <ArrowRight className="w-4 h-4" />
                    <span>{matchedRight.text}</span>
                  </div>
                )}
              </div>
            );
          })}
        </div>

        {/* Right Column */}
        <div className="space-y-3">
          <h3 className="text-sm font-medium text-gray-700">–°—Ç–æ–ª–∏—Ü—ã:</h3>
          {rightItems.map((item) => {
            const isUsed = isRightUsed(item.id);
            const canSelect = selectedLeft !== null;

            return (
              <button
                key={item.id}
                onClick={() => handleRightClick(item.id)}
                disabled={!canSelect}
                className={`w-full p-4 rounded-lg border-2 text-left transition ${
                  isUsed
                    ? 'border-green-300 bg-green-50 opacity-50'
                    : canSelect
                    ? 'border-gray-300 bg-white hover:border-blue-400 hover:bg-blue-50'
                    : 'border-gray-200 bg-gray-50 cursor-not-allowed'
                }`}
              >
                <span className={isUsed ? 'line-through' : ''}>{item.text}</span>
              </button>
            );
          })}
        </div>
      </div>

      <div className="flex items-center justify-between p-4 bg-gray-50 rounded-lg">
        <div className="text-sm">
          –°–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–æ: {Object.keys(matches).length} –∏–∑ {leftItems.length}
          {Object.keys(matches).length === leftItems.length && (
            <span className="ml-4 text-green-600 font-semibold">
              ‚úì –ü—Ä–∞–≤–∏–ª—å–Ω—ã—Ö: {result.correct} –∏–∑ {result.total}
            </span>
          )}
        </div>
        <button
          onClick={() => {
            setMatches({});
            setSelectedLeft(null);
          }}
          className="flex items-center gap-2 text-sm text-gray-600 hover:text-gray-900"
        >
          <RefreshCw className="w-4 h-4" />
          –°–±—Ä–æ—Å–∏—Ç—å
        </button>
      </div>

      <div className="h-2 bg-gray-200 rounded-full overflow-hidden">
        <div
          className="h-full bg-blue-500 transition-all duration-300"
          style={{ width: `${(Object.keys(matches).length / leftItems.length) * 100}%` }}
        />
      </div>
    </div>
  );
};

// ==================== ORDERING QUESTION ====================
const OrderingQuestionDemo = () => {
  const [items, setItems] = useState([
    { id: 1, text: '–ù–∞–±–ª—é–¥–µ–Ω–∏–µ –∏ —Ñ–æ—Ä–º—É–ª–∏—Ä–æ–≤–∞–Ω–∏–µ –≤–æ–ø—Ä–æ—Å–∞', order: 0 },
    { id: 2, text: '–í—ã–¥–≤–∏–∂–µ–Ω–∏–µ –≥–∏–ø–æ—Ç–µ–∑—ã', order: 1 },
    { id: 3, text: '–ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–∞', order: 2 },
    { id: 4, text: '–ê–Ω–∞–ª–∏–∑ –¥–∞–Ω–Ω—ã—Ö', order: 3 },
    { id: 5, text: '–§–æ—Ä–º—É–ª–∏—Ä–æ–≤–∞–Ω–∏–µ –≤—ã–≤–æ–¥–æ–≤', order: 4 },
  ]);
  const [shuffled, setShuffled] = useState(false);

  const handleMoveUp = (index) => {
    if (index === 0) return;
    const newItems = [...items];
    [newItems[index - 1], newItems[index]] = [newItems[index], newItems[index - 1]];
    setItems(newItems);
  };

  const handleMoveDown = (index) => {
    if (index === items.length - 1) return;
    const newItems = [...items];
    [newItems[index], newItems[index + 1]] = [newItems[index + 1], newItems[index]];
    setItems(newItems);
  };

  const handleShuffle = () => {
    const shuffledItems = [...items].sort(() => Math.random() - 0.5);
    setItems(shuffledItems);
    setShuffled(true);
  };

  const isCorrect = items.every((item, index) => item.order === index);

  return (
    <div className="space-y-4">
      <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
        <p className="text-sm text-blue-900">
          üí° –†–∞—Å–ø–æ–ª–æ–∂–∏—Ç–µ —à–∞–≥–∏ –Ω–∞—É—á–Ω–æ–≥–æ –º–µ—Ç–æ–¥–∞ –≤ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ
        </p>
      </div>

      <div className="space-y-2">
        {items.map((item, index) => (
          <div
            key={item.id}
            className={`flex items-center gap-3 p-4 border-2 rounded-lg transition ${
              shuffled && item.order === index
                ? 'border-green-300 bg-green-50'
                : 'border-gray-300 bg-white'
            }`}
          >
            <GripVertical className="w-5 h-5 text-gray-400" />
            
            <div className="w-8 h-8 bg-blue-100 text-blue-600 rounded-full flex items-center justify-center font-semibold flex-shrink-0">
              {index + 1}
            </div>

            <div className="flex-1 font-medium text-gray-900">{item.text}</div>

            <div className="flex gap-1">
              <button
                onClick={() => handleMoveUp(index)}
                disabled={index === 0}
                className="p-2 text-gray-600 hover:bg-gray-100 rounded transition disabled:opacity-30"
              >
                <ArrowUp className="w-4 h-4" />
              </button>
              <button
                onClick={() => handleMoveDown(index)}
                disabled={index === items.length - 1}
                className="p-2 text-gray-600 hover:bg-gray-100 rounded transition disabled:opacity-30"
              >
                <ArrowDown className="w-4 h-4" />
              </button>
            </div>
          </div>
        ))}
      </div>

      <div className="flex items-center justify-between p-4 bg-gray-50 rounded-lg">
        <div className="text-sm">
          {shuffled && (
            <span className={isCorrect ? 'text-green-600 font-semibold' : 'text-gray-600'}>
              {isCorrect ? '‚úì –ü—Ä–∞–≤–∏–ª—å–Ω—ã–π –ø–æ—Ä—è–¥–æ–∫!' : '‚úó –ù–µ–ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –ø–æ—Ä—è–¥–æ–∫'}
            </span>
          )}
        </div>
        <button
          onClick={handleShuffle}
          className="flex items-center gap-2 text-sm text-gray-600 hover:text-gray-900 hover:bg-gray-100 px-4 py-2 rounded-lg"
        >
          <RefreshCw className="w-4 h-4" />
          –ü–µ—Ä–µ–º–µ—à–∞—Ç—å
        </button>
      </div>
    </div>
  );
};

// ==================== HOTSPOT QUESTION ====================
const HotspotQuestionDemo = () => {
  const [clicks, setClicks] = useState([]);

  const correctAreas = [
    { x: 30, y: 40, radius: 8, label: '–°–µ—Ä–¥—Ü–µ' },
    { x: 70, y: 35, radius: 8, label: '–õ—ë–≥–∫–∏–µ' },
  ];

  const handleClick = (e) => {
    const rect = e.currentTarget.getBoundingClientRect();
    const x = ((e.clientX - rect.left) / rect.width) * 100;
    const y = ((e.clientY - rect.top) / rect.height) * 100;

    const newClick = {
      id: Date.now(),
      x: Math.round(x * 100) / 100,
      y: Math.round(y * 100) / 100,
    };

    setClicks([...clicks, newClick]);
  };

  const removeClick = (id) => {
    setClicks(clicks.filter((c) => c.id !== id));
  };

  const checkClick = (click) => {
    return correctAreas.some((area) => {
      const distance = Math.sqrt(
        Math.pow(click.x - area.x, 2) + Math.pow(click.y - area.y, 2)
      );
      return distance <= area.radius;
    });
  };

  const correctClicks = clicks.filter(checkClick).length;

  return (
    <div className="space-y-4">
      <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
        <p className="text-sm text-blue-900">
          üéØ –ö–ª–∏–∫–Ω–∏—Ç–µ –Ω–∞ —Ç–µ–ª–æ —á–µ–ª–æ–≤–µ–∫–∞, —á—Ç–æ–±—ã –æ—Ç–º–µ—Ç–∏—Ç—å: —Å–µ—Ä–¥—Ü–µ –∏ –ª—ë–≥–∫–∏–µ
        </p>
      </div>

      <div
        className="relative border-2 border-gray-300 rounded-lg overflow-hidden bg-gradient-to-b from-blue-100 to-blue-50 cursor-crosshair"
        style={{ height: '400px' }}
        onClick={handleClick}
      >
        {/* Simplified human body */}
        <div className="absolute inset-0 flex items-center justify-center">
          <div className="relative">
            {/* Head */}
            <div className="w-16 h-16 bg-orange-200 rounded-full mx-auto mb-2 border-2 border-orange-300" />
            {/* Body */}
            <div className="w-32 h-40 bg-orange-200 rounded-lg border-2 border-orange-300 relative">
              {/* Heart area hint */}
              <div className="absolute left-6 top-12 w-8 h-8 border-2 border-dashed border-red-300 rounded-full opacity-30" />
              {/* Lungs area hint */}
              <div className="absolute right-6 top-8 w-10 h-10 border-2 border-dashed border-blue-300 rounded-full opacity-30" />
            </div>
            {/* Legs */}
            <div className="flex gap-2 justify-center mt-2">
              <div className="w-12 h-24 bg-orange-200 rounded-lg border-2 border-orange-300" />
              <div className="w-12 h-24 bg-orange-200 rounded-lg border-2 border-orange-300" />
            </div>
          </div>
        </div>

        {/* Render clicks */}
        {clicks.map((click, index) => {
          const isCorrect = checkClick(click);
          return (
            <div
              key={click.id}
              className="absolute transform -translate-x-1/2 -translate-y-1/2 group"
              style={{ left: `${click.x}%`, top: `${click.y}%` }}
            >
              <div className="relative">
                <div
                  className={`w-8 h-8 ${
                    isCorrect ? 'bg-green-500' : 'bg-red-500'
                  } border-4 border-white rounded-full shadow-lg flex items-center justify-center text-white text-xs font-bold`}
                >
                  {index + 1}
                </div>
                
                <button
                  onClick={(e) => {
                    e.stopPropagation();
                    removeClick(click.id);
                  }}
                  className="absolute -top-2 -right-2 w-5 h-5 bg-white border-2 border-red-500 rounded-full text-red-500 opacity-0 group-hover:opacity-100 transition flex items-center justify-center text-xs"
                >
                  √ó
                </button>
              </div>
            </div>
          );
        })}
      </div>

      <div className="flex items-center justify-between p-4 bg-gray-50 rounded-lg">
        <div className="flex items-center gap-2">
          <Target className="w-5 h-5 text-blue-600" />
          <span className="text-sm text-gray-700">
            –û—Ç–º–µ—á–µ–Ω–æ: {clicks.length} —Ç–æ—á–µ–∫
            {clicks.length > 0 && (
              <span className="ml-4 text-green-600 font-semibold">
                ‚úì –ü—Ä–∞–≤–∏–ª—å–Ω—ã—Ö: {correctClicks}
              </span>
            )}
          </span>
        </div>
        <button
          onClick={() => setClicks([])}
          className="flex items-center gap-2 text-sm text-gray-600 hover:text-gray-900"
        >
          <RefreshCw className="w-4 h-4" />
          –°–±—Ä–æ—Å–∏—Ç—å
        </button>
      </div>
    </div>
  );
};

// ==================== FILL BLANKS QUESTION ====================
const FillBlanksQuestionDemo = () => {
  const questionText = "–í—Ç–æ—Ä–∞—è –º–∏—Ä–æ–≤–∞—è –≤–æ–π–Ω–∞ –Ω–∞—á–∞–ª–∞—Å—å –≤ ___ –≥–æ–¥—É, –∫–æ–≥–¥–∞ ___ –Ω–∞–ø–∞–ª–∞ –Ω–∞ –ü–æ–ª—å—à—É. –í–æ–π–Ω–∞ –∑–∞–∫–æ–Ω—á–∏–ª–∞—Å—å –≤ ___ –≥–æ–¥—É.";
  const correctAnswers = {
    0: ['1939'],
    1: ['–≥–µ—Ä–º–∞–Ω–∏—è', 'germany'],
    2: ['1945'],
  };

  const parts = questionText.split(/___/);
  const [blanks, setBlanks] = useState({ 0: '', 1: '', 2: '' });

  const handleChange = (index, value) => {
    setBlanks({ ...blanks, [index]: value });
  };

  const checkBlank = (index) => {
    const userAnswer = blanks[index].toLowerCase().trim();
    return correctAnswers[index].some((ans) => ans.toLowerCase() === userAnswer);
  };

  const filledCount = Object.values(blanks).filter((v) => v.trim()).length;
  const correctCount = Object.keys(blanks).filter((idx) => 
    blanks[idx].trim() && checkBlank(idx)
  ).length;

  return (
    <div className="space-y-4">
      <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
        <p className="text-sm text-blue-900">
          ‚úèÔ∏è –ó–∞–ø–æ–ª–Ω–∏—Ç–µ –ø—Ä–æ–ø—É—Å–∫–∏ –≤ —Ç–µ–∫—Å—Ç–µ
        </p>
      </div>

      <div className="p-6 bg-white border border-gray-300 rounded-lg">
        <div className="text-lg leading-relaxed text-gray-900">
          {parts.map((part, index) => (
            <span key={index}>
              {part}
              {index < parts.length - 1 && (
                <input
                  type="text"
                  value={blanks[index]}
                  onChange={(e) => handleChange(index, e.target.value)}
                  placeholder="___"
                  className={`inline-block mx-1 px-3 py-1 border-b-2 focus:outline-none min-w-[120px] text-center font-medium ${
                    blanks[index].trim() && checkBlank(index)
                      ? 'border-green-500 bg-green-50'
                      : blanks[index].trim()
                      ? 'border-red-500 bg-red-50'
                      : 'border-blue-500 bg-blue-50'
                  }`}
                />
              )}
            </span>
          ))}
        </div>
      </div>

      <div className="p-4 bg-gray-50 rounded-lg">
        <div className="flex items-center justify-between mb-2">
          <div className="flex items-center gap-2 text-sm text-gray-700">
            <FileText className="w-4 h-4" />
            <span>
              –ó–∞–ø–æ–ª–Ω–µ–Ω–æ: {filledCount} –∏–∑ {parts.length - 1}
              {filledCount === parts.length - 1 && (
                <span className="ml-4 text-green-600 font-semibold">
                  ‚úì –ü—Ä–∞–≤–∏–ª—å–Ω—ã—Ö: {correctCount}
                </span>
              )}
            </span>
          </div>
          {filledCount === parts.length - 1 && correctCount === parts.length - 1 && (
            <CheckCircle className="w-5 h-5 text-green-600" />
          )}
        </div>

        <div className="h-2 bg-gray-200 rounded-full overflow-hidden">
          <div
            className="h-full bg-blue-500 transition-all duration-300"
            style={{ width: `${(filledCount / (parts.length - 1)) * 100}%` }}
          />
        </div>
      </div>

      <div className="p-4 bg-yellow-50 border border-yellow-200 rounded-lg text-sm text-yellow-900">
        üí° –ü–æ–¥—Å–∫–∞–∑–∫–∞: 1939, –ì–µ—Ä–º–∞–Ω–∏—è, 1945
      </div>
    </div>
  );
};
