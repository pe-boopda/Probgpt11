import { useState, useEffect } from 'react';
import { Save, X, Plus, Trash2, CheckCircle, Circle } from 'lucide-react';

// –°–∏–º—É–ª—è—Ü–∏—è testService –∏ ImageUploader –¥–ª—è –¥–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏–∏
const testService = {
  getQuestionTypes: () => [
    { value: 'multiple_choice', label: '–û–¥–∏–Ω –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –æ—Ç–≤–µ—Ç' },
    { value: 'multiple_select', label: '–ù–µ—Å–∫–æ–ª—å–∫–æ –ø—Ä–∞–≤–∏–ª—å–Ω—ã—Ö –æ—Ç–≤–µ—Ç–æ–≤' },
    { value: 'true_false', label: '–ü—Ä–∞–≤–¥–∞/–õ–æ–∂—å' },
    { value: 'text_input', label: '–¢–µ–∫—Å—Ç–æ–≤—ã–π –æ—Ç–≤–µ—Ç' },
    { value: 'image_annotation', label: '–ê–Ω–Ω–æ—Ç–∞—Ü–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è' },
    { value: 'matching', label: '–°–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è' },
    { value: 'ordering', label: '–£–ø–æ—Ä—è–¥–æ—á–∏–≤–∞–Ω–∏–µ' },
    { value: 'hotspot', label: '–ì–æ—Ä—è—á–∏–µ —Ç–æ—á–∫–∏' },
    { value: 'fill_blanks', label: '–ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ –ø—Ä–æ–ø—É—Å–∫–æ–≤' },
  ]
};

const ImageUploader = ({ onUpload, currentImageUrl, onRemove }) => (
  <div className="border-2 border-dashed border-gray-300 rounded-lg p-4">
    {currentImageUrl ? (
      <div>
        <img src={currentImageUrl} alt="Preview" className="max-h-48 mx-auto" />
        <button onClick={onRemove} className="mt-2 text-red-600 text-sm">–£–¥–∞–ª–∏—Ç—å</button>
      </div>
    ) : (
      <div className="text-center text-gray-500">
        <input type="file" onChange={(e) => {
          if (e.target.files[0]) {
            onUpload({ id: 'demo-image-id' });
          }
        }} />
      </div>
    )}
  </div>
);

const QuestionEditor = ({ question, onSave, onCancel, saving }) => {
  const [formData, setFormData] = useState({
    question_type: 'multiple_choice',
    question_text: '',
    order: 0,
    points: 1.0,
    image_id: null,
    metadata: null,
    options: [],
  });

  useEffect(() => {
    if (question) {
      setFormData({ ...question });
      if (!question.options || question.options.length === 0) {
        initializeOptions(question.question_type);
      }
    }
  }, [question]);

  const initializeOptions = (type) => {
    let defaultOptions = [];
    
    if (type === 'multiple_choice' || type === 'multiple_select') {
      defaultOptions = [
        { option_text: '', is_correct: false, order: 0 },
        { option_text: '', is_correct: false, order: 1 },
      ];
    } else if (type === 'true_false') {
      defaultOptions = [
        { option_text: '–ü—Ä–∞–≤–¥–∞', is_correct: false, order: 0 },
        { option_text: '–õ–æ–∂—å', is_correct: false, order: 1 },
      ];
    } else if (type === 'matching') {
      defaultOptions = [
        { option_text: '', is_correct: false, order: 0, match_id: 'left_1' },
        { option_text: '', is_correct: false, order: 1, match_id: 'left_2' },
        { option_text: '', is_correct: false, order: 2, match_id: 'right_1' },
        { option_text: '', is_correct: false, order: 3, match_id: 'right_2' },
      ];
    } else if (type === 'ordering') {
      defaultOptions = [
        { option_text: '', is_correct: false, order: 0 },
        { option_text: '', is_correct: false, order: 1 },
        { option_text: '', is_correct: false, order: 2 },
      ];
    } else if (type === 'hotspot') {
      defaultOptions = [];
    }
    
    setFormData((prev) => ({ ...prev, options: defaultOptions }));
  };

  const handleTypeChange = (type) => {
    setFormData((prev) => ({ ...prev, question_type: type }));
    initializeOptions(type);
  };

  const handleOptionChange = (index, field, value) => {
    setFormData((prev) => ({
      ...prev,
      options: prev.options.map((opt, i) =>
        i === index ? { ...opt, [field]: value } : opt
      ),
    }));
  };

  const handleCorrectToggle = (index) => {
    const isSingleChoice = formData.question_type === 'multiple_choice';
    
    setFormData((prev) => ({
      ...prev,
      options: prev.options.map((opt, i) => ({
        ...opt,
        is_correct: isSingleChoice ? i === index : i === index ? !opt.is_correct : opt.is_correct,
      })),
    }));
  };

  // Matching Editor
  const renderMatchingEditor = () => {
    const leftItems = formData.options.filter((opt) => opt.match_id?.startsWith('left_'));
    const rightItems = formData.options.filter((opt) => opt.match_id?.startsWith('right_'));

    const addPair = () => {
      const leftId = `left_${leftItems.length + 1}`;
      const rightId = `right_${rightItems.length + 1}`;
      
      setFormData((prev) => ({
        ...prev,
        options: [
          ...prev.options,
          { option_text: '', is_correct: false, order: prev.options.length, match_id: leftId },
          { option_text: '', is_correct: false, order: prev.options.length + 1, match_id: rightId },
        ],
      }));
    };

    const removePair = (index) => {
      const leftId = `left_${index + 1}`;
      const rightId = `right_${index + 1}`;
      
      setFormData((prev) => ({
        ...prev,
        options: prev.options.filter(
          (opt) => opt.match_id !== leftId && opt.match_id !== rightId
        ),
      }));
    };

    return (
      <div>
        <div className="flex justify-between items-center mb-3">
          <label className="block text-sm font-medium text-gray-700">
            –ü–∞—Ä—ã —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–π *
          </label>
          <button
            type="button"
            onClick={addPair}
            className="flex items-center gap-1 text-sm text-blue-600 hover:text-blue-700"
          >
            <Plus className="w-4 h-4" />
            –î–æ–±–∞–≤–∏—Ç—å –ø–∞—Ä—É
          </button>
        </div>

        <div className="space-y-4">
          {leftItems.map((leftItem, index) => {
            const rightItem = rightItems[index];
            return (
              <div key={index} className="grid grid-cols-2 gap-4 items-start">
                <input
                  type="text"
                  value={leftItem.option_text}
                  onChange={(e) => {
                    const newOptions = [...formData.options];
                    const leftIdx = newOptions.findIndex(
                      (o) => o.match_id === leftItem.match_id
                    );
                    newOptions[leftIdx].option_text = e.target.value;
                    setFormData({ ...formData, options: newOptions });
                  }}
                  placeholder={`–≠–ª–µ–º–µ–Ω—Ç ${index + 1}`}
                  className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                />
                <div className="flex gap-2">
                  <input
                    type="text"
                    value={rightItem?.option_text || ''}
                    onChange={(e) => {
                      const newOptions = [...formData.options];
                      const rightIdx = newOptions.findIndex(
                        (o) => o.match_id === rightItem?.match_id
                      );
                      if (rightIdx !== -1) {
                        newOptions[rightIdx].option_text = e.target.value;
                        setFormData({ ...formData, options: newOptions });
                      }
                    }}
                    placeholder={`–°–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ ${index + 1}`}
                    className="flex-1 px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  />
                  {leftItems.length > 2 && (
                    <button
                      type="button"
                      onClick={() => removePair(index)}
                      className="p-2 text-red-600 hover:bg-red-50 rounded-lg"
                    >
                      <Trash2 className="w-4 h-4" />
                    </button>
                  )}
                </div>
              </div>
            );
          })}
        </div>
        <p className="mt-2 text-sm text-gray-600">
          –°–æ–∑–¥–∞–π—Ç–µ –ø–∞—Ä—ã: —ç–ª–µ–º–µ–Ω—Ç —Å–ª–µ–≤–∞ –∏ –µ–≥–æ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ —Å–ø—Ä–∞–≤–∞
        </p>
      </div>
    );
  };

  // Ordering Editor
  const renderOrderingEditor = () => {
    const addItem = () => {
      setFormData((prev) => ({
        ...prev,
        options: [
          ...prev.options,
          { option_text: '', is_correct: false, order: prev.options.length },
        ],
      }));
    };

    const removeItem = (index) => {
      setFormData((prev) => ({
        ...prev,
        options: prev.options.filter((_, i) => i !== index),
      }));
    };

    return (
      <div>
        <div className="flex justify-between items-center mb-3">
          <label className="block text-sm font-medium text-gray-700">
            –≠–ª–µ–º–µ–Ω—Ç—ã –¥–ª—è —É–ø–æ—Ä—è–¥–æ—á–∏–≤–∞–Ω–∏—è *
          </label>
          <button
            type="button"
            onClick={addItem}
            className="flex items-center gap-1 text-sm text-blue-600 hover:text-blue-700"
          >
            <Plus className="w-4 h-4" />
            –î–æ–±–∞–≤–∏—Ç—å —ç–ª–µ–º–µ–Ω—Ç
          </button>
        </div>

        <div className="space-y-3">
          {formData.options.map((option, index) => (
            <div key={index} className="flex items-center gap-3">
              <div className="w-8 h-8 bg-blue-100 text-blue-600 rounded-full flex items-center justify-center font-semibold text-sm flex-shrink-0">
                {index + 1}
              </div>
              <input
                type="text"
                value={option.option_text}
                onChange={(e) =>
                  handleOptionChange(index, 'option_text', e.target.value)
                }
                placeholder={`–®–∞–≥ ${index + 1}`}
                className="flex-1 px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
              />
              {formData.options.length > 2 && (
                <button
                  type="button"
                  onClick={() => removeItem(index)}
                  className="p-2 text-red-600 hover:bg-red-50 rounded-lg"
                >
                  <Trash2 className="w-4 h-4" />
                </button>
              )}
            </div>
          ))}
        </div>
        <p className="mt-2 text-sm text-gray-600">
          –î–æ–±–∞–≤—å—Ç–µ —ç–ª–µ–º–µ–Ω—Ç—ã –≤ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ. –°—Ç—É–¥–µ–Ω—Ç –±—É–¥–µ—Ç –∏—Ö —É–ø–æ—Ä—è–¥–æ—á–∏–≤–∞—Ç—å.
        </p>
      </div>
    );
  };

  const handleSubmit = () => {
    if (!formData.question_text.trim()) {
      alert('–í–≤–µ–¥–∏—Ç–µ —Ç–µ–∫—Å—Ç –≤–æ–ø—Ä–æ—Å–∞');
      return;
    }

    // Validation for fill_blanks
    if (formData.question_type === 'fill_blanks') {
      const blankCount = (formData.question_text.match(/___/g) || []).length;
      if (blankCount === 0) {
        alert('–î–æ–±–∞–≤—å—Ç–µ —Ö–æ—Ç—è –±—ã –æ–¥–∏–Ω –ø—Ä–æ–ø—É—Å–∫ (___) –≤ —Ç–µ–∫—Å—Ç –≤–æ–ø—Ä–æ—Å–∞');
        return;
      }
    }

    // Validation for hotspot
    if (formData.question_type === 'hotspot' && !formData.image_id) {
      alert('–ó–∞–≥—Ä—É–∑–∏—Ç–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –¥–ª—è –≤–æ–ø—Ä–æ—Å–∞ —Å –≥–æ—Ä—è—á–∏–º–∏ —Ç–æ—á–∫–∞–º–∏');
      return;
    }

    onSave(formData);
  };

  const questionTypes = testService.getQuestionTypes();
  const needsOptions = ['multiple_choice', 'multiple_select', 'true_false'].includes(
    formData.question_type
  );

  return (
    <div className="max-w-4xl mx-auto p-4">
      <div className="bg-white rounded-lg shadow">
        <div className="px-6 py-4 border-b border-gray-200 flex justify-between items-center">
          <h2 className="text-xl font-semibold text-gray-900">
            {question?.id ? '–†–µ–¥–∞–∫—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –≤–æ–ø—Ä–æ—Å–∞' : '–ù–æ–≤—ã–π –≤–æ–ø—Ä–æ—Å'}
          </h2>
          <button onClick={onCancel} className="p-2 hover:bg-gray-100 rounded-lg">
            <X className="w-5 h-5" />
          </button>
        </div>

        <div className="p-6 space-y-6">
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              –¢–∏–ø –≤–æ–ø—Ä–æ—Å–∞ *
            </label>
            <select
              value={formData.question_type}
              onChange={(e) => handleTypeChange(e.target.value)}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
            >
              {questionTypes.map((type) => (
                <option key={type.value} value={type.value}>
                  {type.label}
                </option>
              ))}
            </select>
          </div>

          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              –¢–µ–∫—Å—Ç –≤–æ–ø—Ä–æ—Å–∞ *
            </label>
            <textarea
              value={formData.question_text}
              onChange={(e) =>
                setFormData({ ...formData, question_text: e.target.value })
              }
              placeholder={
                formData.question_type === 'fill_blanks'
                  ? '–í–≤–µ–¥–∏—Ç–µ —Ç–µ–∫—Å—Ç —Å –ø—Ä–æ–ø—É—Å–∫–∞–º–∏. –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ ___ –¥–ª—è –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏—è –ø—Ä–æ–ø—É—Å–∫–æ–≤.'
                  : '–í–≤–µ–¥–∏—Ç–µ –≤–æ–ø—Ä–æ—Å...'
              }
              rows={formData.question_type === 'fill_blanks' ? 6 : 3}
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
            />
            {formData.question_type === 'fill_blanks' && (
              <p className="mt-1 text-sm text-gray-600">
                üí° –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ —Ç—Ä–∏ –ø–æ–¥—á–µ—Ä–∫–∏–≤–∞–Ω–∏—è ___ –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è –ø—Ä–æ–ø—É—Å–∫–æ–≤
              </p>
            )}
          </div>

          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              –ë–∞–ª–ª—ã –∑–∞ –≤–æ–ø—Ä–æ—Å
            </label>
            <input
              type="number"
              value={formData.points}
              onChange={(e) =>
                setFormData({ ...formData, points: parseFloat(e.target.value) || 1 })
              }
              min="0"
              step="0.5"
              className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
            />
          </div>

          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              –ò–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ {formData.question_type === 'hotspot' ? '*' : '(–æ–ø—Ü–∏–æ–Ω–∞–ª—å–Ω–æ)'}
            </label>
            <ImageUploader
              onUpload={(imageData) =>
                setFormData({ ...formData, image_id: imageData.id })
              }
              currentImageUrl={
                formData.image_id ? `/uploads/${formData.image_id}` : null
              }
              onRemove={() => setFormData({ ...formData, image_id: null })}
            />
          </div>

          {/* Choice Questions */}
          {needsOptions && (
            <div className="space-y-3">
              <div className="flex justify-between items-center">
                <label className="block text-sm font-medium text-gray-700">
                  –í–∞—Ä–∏–∞–Ω—Ç—ã –æ—Ç–≤–µ—Ç–æ–≤ *
                </label>
                {formData.question_type !== 'true_false' && (
                  <button
                    onClick={() =>
                      setFormData((prev) => ({
                        ...prev,
                        options: [
                          ...prev.options,
                          { option_text: '', is_correct: false, order: prev.options.length },
                        ],
                      }))
                    }
                    className="flex items-center gap-1 text-sm text-blue-600 hover:text-blue-700"
                  >
                    <Plus className="w-4 h-4" />
                    –î–æ–±–∞–≤–∏—Ç—å –≤–∞—Ä–∏–∞–Ω—Ç
                  </button>
                )}
              </div>

              {formData.options.map((option, index) => (
                <div key={index} className="flex items-start gap-3">
                  <button
                    type="button"
                    onClick={() => handleCorrectToggle(index)}
                    className="mt-2"
                  >
                    {option.is_correct ? (
                      <CheckCircle className="w-5 h-5 text-green-600" />
                    ) : (
                      <Circle className="w-5 h-5 text-gray-400 hover:text-gray-600" />
                    )}
                  </button>
                  <input
                    type="text"
                    value={option.option_text}
                    onChange={(e) =>
                      handleOptionChange(index, 'option_text', e.target.value)
                    }
                    placeholder={`–í–∞—Ä–∏–∞–Ω—Ç ${index + 1}`}
                    className={`flex-1 px-4 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 ${
                      option.is_correct ? 'border-green-300 bg-green-50' : 'border-gray-300'
                    }`}
                    disabled={formData.question_type === 'true_false'}
                  />
                  {formData.question_type !== 'true_false' &&
                    formData.options.length > 2 && (
                      <button
                        type="button"
                        onClick={() =>
                          setFormData((prev) => ({
                            ...prev,
                            options: prev.options.filter((_, i) => i !== index),
                          }))
                        }
                        className="p-2 text-red-600 hover:bg-red-50 rounded-lg mt-1"
                      >
                        <Trash2 className="w-4 h-4" />
                      </button>
                    )}
                </div>
              ))}
            </div>
          )}

          {formData.question_type === 'matching' && renderMatchingEditor()}
          {formData.question_type === 'ordering' && renderOrderingEditor()}

          {/* Info Messages */}
          {formData.question_type === 'hotspot' && (
            <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
              <p className="text-sm text-blue-900">
                üéØ –ó–∞–≥—Ä—É–∑–∏—Ç–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ. –°—Ç—É–¥–µ–Ω—Ç –±—É–¥–µ—Ç –∫–ª–∏–∫–∞—Ç—å –ø–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω—ã–º –æ–±–ª–∞—Å—Ç—è–º.
              </p>
            </div>
          )}

          {formData.question_type === 'fill_blanks' && (
            <div className="p-4 bg-yellow-50 border border-yellow-200 rounded-lg">
              <p className="text-sm text-yellow-900">
                ‚úèÔ∏è –°—Ç—É–¥–µ–Ω—Ç –±—É–¥–µ—Ç –∑–∞–ø–æ–ª–Ω—è—Ç—å –ø—Ä–æ–ø—É—Å–∫–∏ –≤ —Ç–µ–∫—Å—Ç–µ. –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ ___ –¥–ª—è –ø—Ä–æ–ø—É—Å–∫–æ–≤.
              </p>
            </div>
          )}
        </div>

        <div className="px-6 py-4 border-t border-gray-200 flex justify-end gap-3">
          <button
            onClick={onCancel}
            disabled={saving}
            className="px-4 py-2 border border-gray-300 text-gray-700 rounded-lg hover:bg-gray-50 disabled:opacity-50"
          >
            –û—Ç–º–µ–Ω–∞
          </button>
          <button
            onClick={handleSubmit}
            disabled={saving}
            className="flex items-center gap-2 px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50"
          >
            <Save className="w-4 h-4" />
            {saving ? '–°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ...' : '–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –≤–æ–ø—Ä–æ—Å'}
          </button>
        </div>
      </div>
    </div>
  );
};

export default QuestionEditor;
