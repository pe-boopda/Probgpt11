import { useState } from 'react';
import { Download, FileSpreadsheet, FileText, Award, Loader2, CheckCircle } from 'lucide-react';

const ExportButtons = ({ testId, sessionId = null, mode = 'test' }) => {
  const [exporting, setExporting] = useState(null);
  const [success, setSuccess] = useState(null);

  const handleExport = async (format, includeDetails = false) => {
    setExporting(format);
    setSuccess(null);

    try {
      let url;
      
      if (mode === 'test') {
        // Export entire test results
        url = `/api/export/tests/${testId}/export/${format}`;
        if (includeDetails && format !== 'csv') {
          url += '?include_details=true';
        }
      } else if (mode === 'certificate') {
        // Export individual certificate
        url = `/api/export/sessions/${sessionId}/certificate/pdf`;
      }

      const response = await fetch(url, {
        method: 'GET',
        headers: {
          'Authorization': `Bearer ${localStorage.getItem('token')}`,
        },
      });

      if (!response.ok) {
        throw new Error('Export failed');
      }

      // Get filename from headers or use default
      const contentDisposition = response.headers.get('Content-Disposition');
      let filename = `export_${Date.now()}.${format}`;
      
      if (contentDisposition) {
        const matches = /filename="?([^"]+)"?/.exec(contentDisposition);
        if (matches && matches[1]) {
          filename = matches[1];
        }
      }

      // Download file
      const blob = await response.blob();
      const downloadUrl = window.URL.createObjectURL(blob);
      const link = document.createElement('a');
      link.href = downloadUrl;
      link.download = filename;
      document.body.appendChild(link);
      link.click();
      link.remove();
      window.URL.revokeObjectURL(downloadUrl);

      // Show success
      setSuccess(format);
      setTimeout(() => setSuccess(null), 3000);
      
    } catch (error) {
      console.error('Export error:', error);
      alert('–ù–µ —É–¥–∞–ª–æ—Å—å —ç–∫—Å–ø–æ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –¥–∞–Ω–Ω—ã–µ');
    } finally {
      setExporting(null);
    }
  };

  if (mode === 'certificate') {
    return (
      <button
        onClick={() => handleExport('pdf')}
        disabled={exporting}
        className="flex items-center gap-2 px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50 transition"
      >
        {exporting ? (
          <>
            <Loader2 className="w-4 h-4 animate-spin" />
            –°–æ–∑–¥–∞–Ω–∏–µ —Å–µ—Ä—Ç–∏—Ñ–∏–∫–∞—Ç–∞...
          </>
        ) : success ? (
          <>
            <CheckCircle className="w-4 h-4" />
            –°–∫–∞—á–∞–Ω!
          </>
        ) : (
          <>
            <Award className="w-4 h-4" />
            –°–∫–∞—á–∞—Ç—å —Å–µ—Ä—Ç–∏—Ñ–∏–∫–∞—Ç
          </>
        )}
      </button>
    );
  }

  return (
    <div className="flex flex-wrap gap-3">
      {/* Excel Export */}
      <div className="relative group">
        <button
          onClick={() => handleExport('excel', false)}
          disabled={exporting}
          className="flex items-center gap-2 px-4 py-2 bg-green-600 text-white rounded-lg hover:bg-green-700 disabled:opacity-50 transition"
        >
          {exporting === 'excel' ? (
            <>
              <Loader2 className="w-4 h-4 animate-spin" />
              –≠–∫—Å–ø–æ—Ä—Ç...
            </>
          ) : success === 'excel' ? (
            <>
              <CheckCircle className="w-4 h-4" />
              –°–∫–∞—á–∞–Ω!
            </>
          ) : (
            <>
              <FileSpreadsheet className="w-4 h-4" />
              Excel
            </>
          )}
        </button>

        {/* Tooltip */}
        <div className="absolute bottom-full left-0 mb-2 hidden group-hover:block z-10">
          <div className="bg-gray-900 text-white text-xs rounded px-3 py-2 whitespace-nowrap">
            –≠–∫—Å–ø–æ—Ä—Ç –≤ Excel (.xlsx)
            <div className="text-gray-400 mt-1">
              –°–≤–æ–¥–∫–∞ + –¥–µ—Ç–∞–ª—å–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã
            </div>
          </div>
        </div>
      </div>

      {/* Excel with Details */}
      <div className="relative group">
        <button
          onClick={() => handleExport('excel', true)}
          disabled={exporting}
          className="flex items-center gap-2 px-4 py-2 bg-green-700 text-white rounded-lg hover:bg-green-800 disabled:opacity-50 transition"
        >
          {exporting === 'excel-details' ? (
            <>
              <Loader2 className="w-4 h-4 animate-spin" />
              –≠–∫—Å–ø–æ—Ä—Ç...
            </>
          ) : (
            <>
              <FileSpreadsheet className="w-4 h-4" />
              Excel+
            </>
          )}
        </button>

        {/* Tooltip */}
        <div className="absolute bottom-full left-0 mb-2 hidden group-hover:block z-10">
          <div className="bg-gray-900 text-white text-xs rounded px-3 py-2 whitespace-nowrap">
            –†–∞—Å—à–∏—Ä–µ–Ω–Ω—ã–π Excel
            <div className="text-gray-400 mt-1">
              + –∞–Ω–∞–ª–∏–∑ –ø–æ –≤–æ–ø—Ä–æ—Å–∞–º
            </div>
          </div>
        </div>
      </div>

      {/* PDF Export */}
      <div className="relative group">
        <button
          onClick={() => handleExport('pdf', false)}
          disabled={exporting}
          className="flex items-center gap-2 px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 disabled:opacity-50 transition"
        >
          {exporting === 'pdf' ? (
            <>
              <Loader2 className="w-4 h-4 animate-spin" />
              –≠–∫—Å–ø–æ—Ä—Ç...
            </>
          ) : success === 'pdf' ? (
            <>
              <CheckCircle className="w-4 h-4" />
              –°–∫–∞—á–∞–Ω!
            </>
          ) : (
            <>
              <FileText className="w-4 h-4" />
              PDF
            </>
          )}
        </button>

        {/* Tooltip */}
        <div className="absolute bottom-full left-0 mb-2 hidden group-hover:block z-10">
          <div className="bg-gray-900 text-white text-xs rounded px-3 py-2 whitespace-nowrap">
            –≠–∫—Å–ø–æ—Ä—Ç –≤ PDF
            <div className="text-gray-400 mt-1">
              –û—Ç—á—ë—Ç –¥–ª—è –ø–µ—á–∞—Ç–∏
            </div>
          </div>
        </div>
      </div>

      {/* CSV Export */}
      <div className="relative group">
        <button
          onClick={() => handleExport('csv')}
          disabled={exporting}
          className="flex items-center gap-2 px-4 py-2 bg-gray-600 text-white rounded-lg hover:bg-gray-700 disabled:opacity-50 transition"
        >
          {exporting === 'csv' ? (
            <>
              <Loader2 className="w-4 h-4 animate-spin" />
              –≠–∫—Å–ø–æ—Ä—Ç...
            </>
          ) : success === 'csv' ? (
            <>
              <CheckCircle className="w-4 h-4" />
              –°–∫–∞—á–∞–Ω!
            </>
          ) : (
            <>
              <Download className="w-4 h-4" />
              CSV
            </>
          )}
        </button>

        {/* Tooltip */}
        <div className="absolute bottom-full left-0 mb-2 hidden group-hover:block z-10">
          <div className="bg-gray-900 text-white text-xs rounded px-3 py-2 whitespace-nowrap">
            –≠–∫—Å–ø–æ—Ä—Ç –≤ CSV
            <div className="text-gray-400 mt-1">
              –î–ª—è –∞–Ω–∞–ª–∏–∑–∞ –≤ –¥—Ä—É–≥–∏—Ö –ø—Ä–æ–≥—Ä–∞–º–º–∞—Ö
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};
