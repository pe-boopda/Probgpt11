```python
    """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å –æ—Ç–≤–µ—Ç —Ç–∏–ø–∞ hotspot"""

    correct_areas = question_metadata.get('correct_areas', [])
    tolerance = question_metadata.get('tolerance', 5)
    
    matched = 0
    for area in correct_areas:
        for click in clicks:
            distance = math.sqrt(
                (click['x'] - area['x'])**2 + 
                (click['y'] - area['y'])**2

            if distance <= tolerance:
                matched += 1
                break
    
    return matched == len(correct_areas)
```

```python
def check_fill_blanks_answer(self, answer_data, question_metadata):
    """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å –æ—Ç–≤–µ—Ç —Ç–∏–ø–∞ fill_blanks"""
    blanks = answer_data.get('blanks', {})
    correct = question_metadata.get('correct_answers', {})
    case_sensitive = question_metadata.get('case_sensitive', False)
    
    total = len(correct)
    correct_count = 0
    
    for blank_id, user_answer in blanks.items():
        if blank_id in correct:
            expected = correct[blank_id]
            user_ans = user_answer if case_sensitive else user_answer.lower()
            
            # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ —Å –ª—é–±—ã–º –∏–∑ –ø—Ä–∞–≤–∏–ª—å–Ω—ã—Ö –≤–∞—Ä–∏–∞–Ω—Ç–æ–≤
            for exp in expected:
                exp_ans = exp if case_sensitive else exp.lower()
                if user_ans.strip() == exp_ans.strip():
                    correct_count += 1
                    break
    
    return correct_count == total
```
